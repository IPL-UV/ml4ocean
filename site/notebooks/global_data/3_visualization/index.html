ax = plt.axes(projection=ccrs.Mollweide())

pts1 = plt.scatter(
    x=soca2016_locations.lon, 
    y=soca2016_locations.lat, 
    s=80,
    marker='+',
    color="darkorange",
    transform=ccrs.PlateCarree(),
    label='SOCA2016', zorder=3)
# ISPRS2020
pts2 = plt.scatter(
    x=isprs2020_locations.lon, 
    y=isprs2020_locations.lat,

    transform=ccrs.PlateCarree(),

pts3 = plt.scatter(
